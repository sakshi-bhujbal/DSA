#include<stdio.h>
#include<limits.h>

    void findMinMax(int a[],int N)
    {
        int minE=INT_MAX;
        int maxE=INT_MIN;
        
        for(int i=0;i<N;i++)
        {
            if(a[i]<minE)
               { 
                   minE=a[i];
               }
            if(a[i]>maxE)
              {
                 maxE=a[i];
              }
         }
    

  printf("The minimum number is:%d",minE);
  printf("\nThe maximum number is:%d",maxE);
  
  return ;
    }
  
  
  int main()
  {
      int a[]={1,4,6,8,9};
      
      int N = sizeof(a)/sizeof(a[0]);
      
      findMinMax(a,N);
      return 0;
  }

int main()
{
    int a[5]={1,4,7,9,3};
    
    printf("Before reversing the array elements are:\n");
    for(int i=0;i<5;i++)
    {
        printf("a[%d]=%d\n",i,a[i]);
    }
    
    printf("After reversing the array elements are:");
    for(int i=5;i>=0;i--)
    {
        printf("a[%d]=%d\n",i,a[i]);
    }
    
}
